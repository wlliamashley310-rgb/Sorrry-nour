<script>
const message = `I'm really sorry for how I acted last night.
My behavior was completely unacceptable,
and I truly regret it.

I know I hurt you, and I don't expect
everything to be okay immediately.
I just wanted you to know that I'm genuinely sorry.

No matter what happens,
thank you for reading this. 💗`;

const envelope = document.getElementById('envelope');
const letter = document.getElementById('letter');
const textBox = document.getElementById('text');
const hearts = document.getElementById('hearts');
const petals = document.getElementById('petals');
const sparkles = document.getElementById('sparkles');
const footer = document.getElementById('footer');

let opened = false;

function createStars(){
    const stars = document.getElementById('stars');
    for(let i=0;i<90;i++){
        const s = document.createElement('div');
        s.className = 'star';
        s.style.left = Math.random()*100 + 'vw';
        s.style.top = Math.random()*100 + 'vh';
        s.style.animationDelay = Math.random()*2 + 's';
        stars.appendChild(s);
    }
}

function createHeartBurst(){
    const rect = envelope.getBoundingClientRect();
    const originX = rect.left + rect.width/2;
    const originY = rect.top + rect.height/2;

    for(let i=0;i<70;i++){
        const h = document.createElement('div');
        h.className = 'heart';
        h.textContent = Math.random() > .5 ? '💗' : '💕';

        const dx = (Math.random()-0.5)*260 + 'px';
        h.style.setProperty('--x', originX + 'px');
        h.style.setProperty('--y', originY + 'px');
        h.style.setProperty('--dx', dx);

        h.style.fontSize = (18 + Math.random()*22) + 'px';
        h.style.animationDelay = Math.random()*0.5 + 's';

        hearts.appendChild(h);
        setTimeout(()=>h.remove(),6500);
    }
}

function createPetals(){
    let count = 0;
    const timer = setInterval(()=>{
        count++;
        const p = document.createElement('div');
        p.className = 'petal';
        p.textContent = Math.random() > .5 ? '🌸' : '✨';
        p.style.left = Math.random()*100 + 'vw';
        p.style.fontSize = (16 + Math.random()*16) + 'px';
        p.style.animationDuration = (5 + Math.random()*6) + 's';
        petals.appendChild(p);
        setTimeout(()=>p.remove(),12000);

        if(count > 30) clearInterval(timer);
    },250);
}

function createSparkles(){
    for(let i=0;i<18;i++){
        const s = document.createElement('div');
        s.className = 'sparkle';
        s.textContent = Math.random() > .5 ? '✨' : '💫';
        s.style.left = 25 + Math.random()*50 + '%';
        s.style.top = 20 + Math.random()*55 + '%';
        s.style.animationDelay = Math.random()*0.8 + 's';
        sparkles.appendChild(s);
        setTimeout(()=>s.remove(),2000);
    }
}

function typeMessage(){
    let i = 0;
    const timer = setInterval(()=>{
        textBox.textContent += message.charAt(i);
        i++;
        if(i >= message.length){
            clearInterval(timer);
            footer.classList.add('show');
        }
    },38);
}

envelope.addEventListener('click', ()=>{
    if(opened) return;
    opened = true;

    envelope.classList.add('open');

    setTimeout(()=>{
        letter.classList.add('show');
        createHeartBurst();
        createSparkles();
        createPetals();

        setTimeout(typeMessage,700);
    },700);
});

createStars();
</script>
